CREATE SCHEMA GPManager;

SET search_path TO GPManager; -- Usado para referenciar a qual pasta de tabelas estamos utilizando

CREATE TABLE Pais(
	ID SERIAL NOT NULL,
	NOME VARCHAR(100) NOT NULL,
	ISO CHAR(2) NOT NULL, 
	PRIMARY KEY(ID)
);

CREATE TABLE Piloto(
    ID INTEGER NOT NULL,
    ID_PAIS INTEGER NOT NULL, 
    NOME VARCHAR(50) NOT NULL,
    IDADE INTEGER NOT NULL,
    STATUS VARCHAR(20),
    LA INTEGER CHECK (LA BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    CO INTEGER CHECK (CO BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    UL INTEGER CHECK (UL BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    EX INTEGER CHECK (EX BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    RA INTEGER CHECK (RA BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    CH INTEGER CHECK (CH BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    AC INTEGER CHECK (AC BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    FI INTEGER CHECK (FI BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    POT INTEGER CHECK (POT BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    AUGE INTEGER NOT NULL DEFAULT 0,
    APOS BOOLEAN NOT NULL DEFAULT 0,
    TITULOS INTEGER NOT NULL DEFAULT 0,
    VITORIAS INTEGER NOT NULL DEFAULT 0,
    POLES INTEGER NOT NULL DEFAULT 0,
    PODIOS INTEGER NOT NULL DEFAULT 0,
    GPs INTEGER NOT NULL DEFAULT 0,
    PRIMARY KEY (ID),
    FOREIGN KEY(ID_PAIS) REFERENCES Pais(ID)
);

CREATE TABLE Circuito(
	ID INTEGER NOT NULL,
	NUM INTEGER NOT NULL,
	ID_PAIS INTEGER NOT NULL, 
	PROVA VARCHAR(100) NOT NULL,
	CIRCUITO VARCHAR(100) NOT NULL,
	RETAS INTEGER CHECK (RETAS BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
	CURVAS INTEGER CHECK (CURVAS BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
	VOLTAS INTEGER NOT NULL,
	TEMPO_BASE TIME(3) NOT NULL,
	PRIMARY KEY (ID),
	FOREIGN KEY(ID_PAIS) REFERENCES Pais(ID)
);

CREATE TABLE Motor(
	ID SERIAL NOT NULL,
	ID_PAIS_MOTOR INTEGER NOT NULL, 
	FORNECEDORA VARCHAR(100) NOT NULL,
	POTENCIA INTEGER NOT NULL,
	PRIMARY KEY(ID),
	FOREIGN KEY(ID_PAIS_MOTOR) REFERENCES PAIS(ID)
);

CREATE TABLE Equipe(
    ID INTEGER NOT NULL,
    ID_PAIS INTEGER NOT NULL, 
    NOME VARCHAR(100) NOT NULL,	
    COR1 CHAR(7) NOT NULL DEFAULT '#FFFFFF',
    COR2 CHAR(7) NOT NULL DEFAULT '#FFFFFF',
    PILOTO1 INTEGER,
    PILOTO2 INTEGER,
    PILOTOTESTES INTEGER,
    AERO INTEGER CHECK (AERO BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    CAMBIO INTEGER CHECK (CAMBIO BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    FREIO INTEGER CHECK (FREIO BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    ELETR INTEGER CHECK (ELETR BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    DIREC INTEGER CHECK (DIREC BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    CONFI INTEGER CHECK (CONFI BETWEEN 0 AND 100) NOT NULL DEFAULT 50,
    ULT_ANO INTEGER,
    ID_MOTOR INTEGER NOT NULL,
    PROPRIO BOOLEAN,
    FOREIGN KEY(ID_PAIS) REFERENCES Pais(ID),
    FOREIGN KEY(ID_MOTOR) REFERENCES Motor(ID),
    FOREIGN KEY(PILOTO1) REFERENCES Piloto(ID),
    FOREIGN KEY(PILOTO2) REFERENCES Piloto(ID),
    FOREIGN KEY(PILOTOTESTES) REFERENCES Piloto(ID),
    PRIMARY KEY(ID)
);

ALTER TABLE Piloto
ADD COLUMN EQUIPE INTEGER,
ADD FOREIGN KEY (EQUIPE) REFERENCES Equipe(ID);

CREATE TABLE HallDaFama(
	ID INTEGER NOT NULL,
	ID_PAIS INTEGER NOT NULL,
	NOME VARCHAR(100) NOT NULL,
	Titulos INTEGER NOT NULL DEFAULT 0,
	Vitoria INTEGER NOT NULL DEFAULT 0,
	Poles INTEGER NOT NULL DEFAULT 0,
	GPs INTEGER NOT NULL DEFAULT 0,
	PRIMARY KEY(ID),
	FOREIGN KEY(ID_PAIS) REFERENCES Pais(ID)
);
